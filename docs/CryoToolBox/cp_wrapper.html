134class ThermState:
135    def __init__(self, fluid, backend="HEOS", **state_parameters):
136        """
137        Available backends: HEOS (opensource), REFPROP.
138        See http://www.coolprop.org/coolprop/REFPROP.html for details.
139        """
140        # TODO this should check for the fluid name too
141        if backend == "HEPROP":
142            if fluid.lower() != 'helium':
143                raise ValueError(f'Only helium fluid can be used with HEPROP backend: {fluid}')
144            self._AbstractState = HepropState()
145        else:
146            self._AbstractState = CP.AbstractState(backend, fluid)
147        if state_parameters:
148            self.update_kw(**state_parameters)
149
150    def update_kw(self, **state_parameters):
151        """Update thermodynamic state using keyword arguments."""
152        if len(state_parameters) != 2:
153            raise TypeError(f'update_kw() takes 2 arguments \
154            ({len(state_parameters)} given)')
155        # TODO Code below can be simplified
156        names = []
157        values = []
158        for name, value in state_parameters.items():
159            names.append(name)
160            values.append(value)
161        self.update(names[0], values[0],
162                    names[1], values[1])
163
164    def update(self, name1, value1, name2, value2):
165        if name1 > name2:  # Sorting inputs alphabetically
166            name1, name2 = name2, name1
167            value1, value2 = value2, value1
168        CP_input_str = name1 + name2
169        CP_value1 = self.prepare_input(name1, value1)
170        CP_value2 = self.prepare_input(name2, value2)
171
172        if self.backend == "HEPROP":
173            self._AbstractState.update(name1, CP_value1, name2, CP_value2)
174        else:
175            self._AbstractState.update(
176                CP_inputs[CP_input_str], CP_value1, CP_value2) ###update all properties in coolprop

179    def prepare_input(name, value):
180        """Prepare value to input to CoolProp.
181
182        Converts the value to units expected by CoolProp
183        (see `CP_const_unit`).
184
185        Parameters
186        ----------
187        name : str
188            Parameter name.
189        value : ureg.Quantity
190            Parameter value.
191
192        Returns
193        -------
194        float
195        """
196        if isinstance(value, (int, float)):
197            return value  # Dimensionless
198        else:
199            unit = CP_const_unit[name][1]
200            return value.to(unit).magnitude

442    @property
443    def specific_heat_input(self):
444        """
445        Calculate Specific heat input, v * (dh/dv)|p.
446        This function is not described in AbstractState class.
447        """
448        # Because specific volume is not available as function of
449        # the AbstractState, density is used instead
450        # The resulting function is: -Dmass*(dHmass/dDmass)|p
451        if self.backend == "HEPROP":
452            return self._AbstractState.specific_heat_input() * ureg.joule/ureg.kg
453        else:
454            return (-self.Dmass) * self.first_partial_deriv('Hmass', 'Dmass', 'P')

456    @property
457    def gamma(self):
458        """Calculate gamma = k = Cp/Cv coefficient.
459        """
460        # To avoid real gas effects influencing Cp and Cv,
461        # calculating at gamma at NTP
462        T_current = self.T
463        S_current = self.Smass
464        self.update('P', P_NTP, 'T', T_NTP)
465        _gamma = self.Cpmass / self.Cvmass
466        self.update('T', T_current, 'Smass', S_current)
467        return _gamma

469    @property
470    def C_gas_const(self):
471        """
472        Constant for gas or vapor which is the function of the ratio of
473        specific heats k = Cp/Cv. ASME VIII.1-2015 pp. 423-424.
474        """
475        k_ = self.gamma.magnitude
476        root = (k_ * (2/(k_+1))**((k_+1)/(k_-1)))**0.5
477        R = ureg.R  # Universal gas constant
478        C_ = root / R**0.5 * (ureg.g/ureg.mol)**0.5
479        return C_
