  29def to_scfma(M_dot_fluid, fluid):
  30    """
  31    Convert mass flow rate into equivalent flow of air.
  32    Flow through a relief device with invariant Area/discharge coefficient
  33    (KA).
  34
  35    Parameters
  36    ----------
  37    M_dot_fluid : Quantity {mass: 1, time: -1}
  38        mass flow rate
  39    fluid : ThermState
  40
  41    Returns
  42    -------
  43    ThermState
  44        volumetric air flow rate
  45    """
  46    C_fluid = fluid.C_gas_const
  47    C_air = AIR.C_gas_const
  48
  49    # Calculation
  50    M_dot_air = M_dot_fluid * C_air / C_fluid * AIR.MZT / fluid.MZT
  51    Q_air = M_dot_air / AIR.Dmass
  52    Q_air.ito(ureg.ft**3/ureg.min)
  53    return Q_air

  56def from_scfma(Q_air, fluid):
  57    """
  58    Convert volumetric air flow rate into equivalent mass flow of specified
  59    fluid. Flow through a relief device with invariant Area/discharge
  60    coefficient (KA).
  61    Invert function to to_scfma().
  62
  63    Parameters
  64    ----------
  65    Q_air : Quantity {length: 3, time: -1}
  66        volumetric air flow rate
  67    fluid : ThermState
  68
  69    Returns
  70    -------
  71    Quantity {mass: 1, time: -1}
  72        mass flow rate
  73    """
  74    C_fluid = fluid.C_gas_const
  75    C_air = AIR.C_gas_const
  76
  77    # Calculation
  78    M_dot_air = Q_air * AIR.Dmass
  79    M_dot_fluid = M_dot_air * C_fluid / C_air * fluid.MZT / AIR.MZT
  80    M_dot_fluid.ito(ureg.g/ureg.s)
  81    return M_dot_fluid

  84def to_standard_flow(flow_rate, fluid):
  85    '''
  86    Converting volumetric flow at certain conditions or mass flow to
  87    flow at NTP.
  88    '''
  89    fluid_NTP = ThermState(fluid.name)
  90    fluid_NTP.update('T', T_NTP, 'P', P_NTP)
  91    if flow_rate.dimensionality == ureg('kg/s').dimensionality:
  92        # mass flow, flow conditions are unnecessary
  93        q_std = flow_rate / fluid_NTP.Dmass
  94    elif flow_rate.dimensionality == ureg('m^3/s').dimensionality:
  95        # volumetric flow given, converting to standard pressure and
  96        # temperature
  97        if fluid.Dmass != -float('Inf')*ureg.kg/ureg.m**3:
  98            # By default ThermState is initialized with all fields == -inf
  99            q_std = flow_rate * fluid.Dmass / fluid_NTP.Dmass
 100        else:
 101            logger.warning('''Flow conditions for volumetric flow {:.3~}
 102                           are not set. Assuming standard flow at NTP.
 103                           '''.format(flow_rate))
 104            q_std = flow_rate
 105    else:
 106        raise ValueError ('Flow dimensionality is not supported: '
 107                     f'{flow_rate.dimensionality}.')
 108    q_std.ito(ureg.ft**3/ureg.min)
 109    return q_std

 141def A_relief_API(m_dot, fluid, *, P_back=P_NTP, K_d=0.975, K_b=1, K_c=1):
 142    """Calculate required relief area for given flow as per
 143    API 520 5.6.3/4
 144
 145    Parameters
 146    ----------
 147    m_dot : Quantity {mass:1, time:-1}
 148        mass flow rate
 149    fluid : ThermState at relief conditions
 150    P_back : ureg.Quantity {length: -1, mass: 1, time: -2}
 151        backpressure
 152    K_d : discharge coefficient
 153        0.975 - when PRV installed with/without a rupture disk
 154        0.62 - for rupture disc only (see 5.11.1.1.2)
 155    K_c : capacity correction factor due to backpressure
 156        applies to balanced bellows valves only
 157    K_c : combination correction factor
 158        1 - for no rupture disc installed in combination
 159        0.9 - for rupture disc installed in combination
 160    """
 161    W = m_dot.m_as(ureg.lb/ureg.hr)
 162    P_1 = fluid.P.m_as(ureg.psi)
 163    P_2 = P_back.m_as(ureg.psi)
 164    k = fluid.gamma
 165    T = fluid.T.m_as(ureg.degR)
 166    Z = fluid.compressibility_factor
 167    M = fluid.M
 168    P_cf = P_1 * (2/(k+1))**(k/(k-1))
 169    if P_2 <= P_cf:
 170        critical = True  # For future verbose use
 171        C = fluid.C.m_as(ureg.lb/(ureg.hr*ureg.lbf)*(ureg.degR)**0.5)
 172        A = W / (C*K_d*P_1*K_b*K_c) * (T*Z/M)**0.5
 173    else:
 174        critical = False
 175        r = P_2 / P_1
 176        F_2_brack = (1-r**((k-1)/k)) / (1-r)
 177        F_2 = (k/(k-1) * r**(2/k) * F_2_brack)**0.5
 178        A = W / (735*F_2*K_d*K_c) * (T*Z/(M*P_1*(P_1-P_2)))**0.5
 179    return A * ureg.inch**2

 219def rad_hl(T_1, eps_1, T_2, eps_2, F1_2=1, baffles={'N': 0, 'eps': 0.02}):
 220    """
 221    Calculate radiative heat load including reduction due to baffles.
 222    Based on Kaganer "Thermal insulation in cryogenic engineering", p. 42.
 223
 224    Parameters
 225    ----------
 226    eps_1 : float
 227        emissivity of the first surface
 228    eps_2 : float
 229        emissivity of the second surface
 230    T_1 : Quantity {temperature: 1}
 231        temperature of the first surface
 232    T_2 : Quantity {temperature: 1}
 233        temperature of the second surface
 234    F1_2 : float
 235        F1_2 = F_cold/F_hot
 236    baffles : dict
 237        N - number of baffles
 238        eps - emissivity of the baffle, assumed to be same on both sides
 239
 240    Returns
 241    -------
 242    dict
 243            :q0: heat load without any baffles
 244            q_baffle : heat load with the baffles
 245            eta : effectiveness of the baffles
 246    """
 247    # TODO This function will be refactored
 248    N_baffles = baffles['N']
 249    eps_baffle = baffles['eps']
 250
 251    eps_mut = 1/(1/eps_1 + F1_2*(1/eps_2-1))  # Mutual emissivity
 252    T_1.ito(ureg.K)
 253    T_2.ito(ureg.K)
 254    q0 = eps_mut*sigma*(T_2**4 - T_1**4)*F1_2
 255    eps_baffle_mut = eps_baffle/(2-eps_baffle)
 256    eta = (1+N_baffles*eps_mut/eps_baffle_mut)**(-1)
 257    q_baffle = eta*q0
 258    return q_baffle.to(ureg.W/ureg.m**2)

 285def conduction_cyl(D_i, D_o, L, k, dT):
 286    """Calculate heat flow through cylindrical wall
 287
 288    Parameters
 289    ----------
 290    D_i : Quantity {length: 2}

 310def Re(fluid, m_dot, D_H, A_cross):
 311    """
 312    Calculate Reynolds number.
 313
 314    Parameters
 315    ----------
 316    fluid : ThermState object describing thermodynamic state (fluid, T, P)
 317    m_dot : mass flow
 318    D_H : hydraulic diameter of a pipe
 319    A_cross : cross-sectional area of a pipe
 320
 321    Returns
 322    -------
 323    Reynolds number, dimensionless
 324    """
 325    rho = fluid.Dmass
 326    v = m_dot / (rho*A_cross)
 327    mu = fluid.viscosity
 328    Re_ = v * D_H * rho / mu
 329    return float(Re_)

 347def Gr(fluid, T_surf, L_surf):
 348    """
 349    Calculate Grashof number.
 350
 351    Parameters
 352    ----------
 353    fluid : ThermState object describing thermodynamic state (fluid, T, P)
 354    T_surf : surface temperature
 355    L_surf : characteristic length
 356
 357    Returns
 358    -------
 359    Grashof number, dimensionless
 360    """
 361    nu_fluid = fluid.viscosity/fluid.Dmass  # kinematic viscosity
 362    beta_exp = fluid.isobaric_expansion_coefficient
 363    Gr_ = ureg.g_0 * L_surf**3 * beta_exp * abs(T_surf-fluid.T) / nu_fluid**2
 364    return float(Gr_)

 404def Kn(fluid, L):
 405    """
 406    Calculate Knudsen number.
 407
 408    Parameters
 409    ----------
 410    fluid : ThermState object describing thermodynamic state (fluid, T, P)
 411    L : characteristic length
 412
 413    Returns
 414    -------
 415    Knudsen number, dimensionless
 416
 417    Notes
 418    -----
 419    The Knudsen number can be used to determine the flow regime:
 420    $Kn < 0.01$: Continuum flow

 425    lambda_ = mean_free_path(fluid)

 430def Nu_blend(Nu_lam, Nu_turb, m):
 431    """Calculate Nu number using blending equation of Churchill and Usagi.
 432    From Handbook of heat transfer by Rohsenow, Hartnet, Cho (HHT) (4.32).

 439    From Handbook of heat transfer by Rohsenow, Hartnet, Cho (HHT) (4.24).

 458def Nu_l_vplate(Pr_, Ra_):
 459    """Calculate laminar Nu number for a vertical isothermal plate.

 488def Nu_l_hcyl(Pr_, Ra_):
 489    """Calculate laminar Nusselt number for isothermal horizontal cylinder.
 490    From Handbook of heat transfer by Rohsenow, Hartnet, Cho (HHT) (4.45b).
 491    """
 492    if Ra_ >= 1e-4:
 493        f = 0.8
 494    else:
 495        f = 1 - 0.13/Nu_T_hcyl(Pr_, Ra_)**0.16
 496    return 2*f / log(1+2*f/Nu_T_hcyl(Pr_, Ra_))

 506def C_t_bar_cyl(Pr_):
 507    """Interpolate value of the turbulent coefficient C_t for isothermal
 508    horizontal cylinder (C/L=1).
 509    From Handbook of heat transfer by Rohsenow, Hartnet, Cho (HHT) Table 4.2.
 510    """
 511    # Using 0.71 value for 0.7..0.71 Pr rang

 813def nist_property(material, prop, T1, T2=None, RRR_OFHC=None):
 814    """
 815    Calculate specific heat capacity using NIST properties database.
 816    https://trc.nist.gov/cryogenics/materials/materialproperties.htm
 817
 818    Parameters
 819    ----------
 820    T1 : Quantity, temperature
 821    T2 : Quantity, temperature
 822        used for average value calculation
 823
 824    Returns
 825    -------
 826    specific heat capacity
 827    """
 828    if material == Material.OFHC and prop == Property.TC:
 829        if RRR_OFHC is None:
 830            logger.warning('RRR for OFHC is not defined. Using RRR=100.')
 831            RRR_OFHC = 100
 832        try:
 833            coefs = _NIST_DATA[material][prop]['coefs'+str(RRR_OFHC)]
 834        except KeyError:
 835            raise NISTError(f'No data for RRR {RRR_OFHC}. '
 836                            'Use RRR 50, 100, 150, 300 or 500')
 837    else:
 838        coefs = _NIST_DATA[material][prop]['coefs']
 839    fun = _NIST_DATA[material][prop]['fun']
 840    unit = _NIST_DATA[material][prop]['unit']
 841    eq_range = _NIST_DATA[material][prop]['range']
 842
 843    T1 = T1.to(ureg.K).magnitude
 844    if prop == Property.LE:
 845        Tlow = _NIST_DATA[material][prop]['Tlow']
 846        if T1 < Tlow:
 847            f = _NIST_DATA[material][prop]['f']
 848            return f * unit
 849    if T1 < eq_range[0] or T1 > eq_range[1]:
 850        raise ValueError(f'Temperature is out of bounds: {T1} for'
 851                         f' {eq_range[0]}-{eq_range[1]}'
 852                         'limits.')
 853    if T2 is None:
 854        value = fun(T1, coefs)
 855    else:
 856        T2 = T2.to(ureg.K).magnitude
 857        if prop == Property.LE:
 858            value_1 = nist_property(material, prop, T1*ureg.K)
 859            value_2 = nist_property(material, prop, T2*ureg.K)
 860            value = value_2 - value_1
 861            return value
 862        else:
 863            value = _nist_quad(T1, T2, fun, coefs)
 864    return value * unit
