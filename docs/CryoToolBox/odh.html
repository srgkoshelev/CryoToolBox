 104    def add_pipe_failure(self, tube, fluid, q_std_rupture=None, N_welds=1):
 105        """Add pipe failure to the leaks dict.
 106
 107        For a given tube calculate leak parameters as following:
 108        - failure rate
 109        - standard volumetric flow
 110        - duration of the release
 111        - number of possible similar events
 112        for all failure modes for piping and welds listed in Table 2 of
 113        FESHM 4240.
 114
 115        Failure modes are analyzed by `Volume.odh` method.
 116
 117        Parameters
 118        ----------
 119        tube : heat_transfer.piping.Tube
 120        fluid : heat_transfer.ThermState
 121            Thermodynamic state of the fluid for the release.
 122        q_std_rupture : ureg.Quantity, [length]^3/[time]
 123            Standard volumetric flow rate for pipe rupture.
 124        N_welds : int
 125            Number of welds on the tube.
 126        """
 127        # Failure rate coefficients; Piping failure rate is per unit of length,
 128        # weld is dependent on number of welds, pipe OD and wall thickness
 129        failure_rate_coeff = {'Piping': (tube.L, 1),
 130                              'Pipe weld': (tube.OD / tube.wall,
 131                                            N_welds)}
 132        # Piping and weld leaks as per Table 2
 133        for cause, fr_coefs in failure_rate_coeff.items():
 134            for mode in TABLE_2[cause].keys():
 135                if tube.D > 2 or mode != 'Large leak':  # Large leak only for D > 2"
 136                    name = f'{cause} {mode.lower()}: {tube}'
 137                    fr_coef, N_events = fr_coefs
 138                    if fr_coef.magnitude == 0:
 139                        raise ODHError('Failure rate should not be 0 '
 140                                       f'{cause} {mode}.')
 141                    if N_events == 0:
 142                        continue
 143                    if mode == 'Rupture':
 144                        failure_rate = fr_coef * TABLE_2[cause][mode]
 145                        if q_std_rupture is not None:
 146                            q_std = q_std_rupture
 147                        else:
 148                            area = tube.area
 149                            q_std = hole_leak(area, fluid)
 150                    else:
 151                        failure_rate = fr_coef * \
 152                            TABLE_2[cause][mode]['Failure rate']
 153                        area = TABLE_2[cause][mode]['Area']
 154                        if area > tube.area:
 155                            ODHError('Leak area cannot be larger'
 156                                     ' than pipe area.')
 157                        q_std = hole_leak(area, fluid)
 158                    self.add_failure_mode(name, failure_rate, fluid, q_std,
 159                                          N_events)

 178    def add_flange_failure(self, pipe, fluid, q_std_rupture=None, N=1,
 179                           gasket_type='reinforced', blowout_area=None):
 180        """Add reinforced or preformed gasket flange failure
 181        to leaks dict.
 182
 183        Store failure rate, flow rate and expected time duration of
 184        the event for transfer line failure. Based on FESHM 4240.
 185        Failure modes are analyzed by `Volume.odh` method.
 186
 187        Parameters
 188        ----------
 189        pipe : heat_transfer.Pipe
 190        fluid : heat_transfer.ThermState
 191            Thermodynamic state of the fluid stored in the source.
 192        q_std_rupture : ureg.Quantity, [length]^3/[time]
 193            Standard volumetric flow rate for flange rupture.
 194        N : int
 195            Number of reinforced seal connections on the pipe.
 196        gasket_type : str
 197            Type of the gasket in the flange, 'soft' or 'reinforced'.
 198        blowout_area : Quantity [length^2]
 199            Leak area in case of the soft gasket blowout. Not applicable for reinforced gaskets.
 200        """
 201        # TODO Make leak and rupture areas adjustable, add info to docstring
 202        if gasket_type == 'reinforced':
 203            table = TABLE_2['Flange, reinforced gasket']
 204        if gasket_type == 'soft':
 205            table = TABLE_2['Flange, soft gasket']
 206        # Leak case
 207        name = f'Flange {gasket_type} gasket leak: {pipe}'
 208        failure_rate = table['Leak']['Failure rate']
 209        area = table['Leak']['Area']
 210        q_std = hole_leak(area, fluid)
 211        self.add_failure_mode(name, failure_rate, fluid, q_std, N)
 212        # Blowout for soft gasket
 213        if gasket_type == 'soft':
 214            name = f'Flange soft gasket blowout: {pipe}'
 215            failure_rate = table['Blowout']
 216            area = blowout_area or pipe.area
 217            q_std = hole_leak(area, fluid)
 218            if q_std_rupture is not None:
 219                # Blowout flow cannot be greater than rupture flow
 220                q_std = min(q_std, q_std_rupture)
 221            self.add_failure_mode(name, failure_rate, fluid, q_std, N)
 222        # Rupture
 223        name = f'Flange {gasket_type} gasket rupture: {pipe}'
 224        failure_rate = table['Rupture']
 225        if q_std_rupture is not None:
 226            q_std = q_std_rupture
 227        else:
 228            area = pipe.area
 229            q_std = hole_leak(area, fluid)
 230        self.add_failure_mode(name, failure_rate, fluid, q_std, N)

 402    def add_failure_mode(self, name, failure_rate, fluid, q_std, N=1):
 403        """Add general failure mode to leaks dict.
 404
 405        Store failure rate, flow rate and expected time duration of the
 406        failure event for general failure mode.
 407        Failure modes are analyzed by `Volume.odh` method.
 408
 409        Parameters
 410        ----------
 411        name : str
 412            Name of the failure mode
 413        failure rate : ureg.Quantity, 1/[time]

 421            Number of similar failure modes.

 423        N_events = N * self.N

 433    def combine(name, fluid, sources, N=1, PFD_isol_valve=1):
 434        """Combine several ODH sources sharing volume.
 435
 436        Can be used for failure modes affecting several sources in parallel.
 437
 438        Parameters
 439        ----------
 440        name : str
 441            Name of the new combined source.
 442        fluid : heat_transfer.ThermState
 443            Thermodynamic state of the fluid stored in the source.
 444        sources : list of Source
 445            Sources connected together.
 446        N : int
 447            Number of the sources if several similar sources exist,
 448            e.g. gas bottles.
 449        PFD_isol_valve : float
 450            Probability of failure on demand of source isolation valve.
 451            If no isolation valve will trigger on ODH alarm, the probability
 452            is 1 (default).
 453
 454        Returns
 455        -------
 456        Source
 457            Combined source of inert gas.
 458        """
 459        if not all([source.fluid.name == fluid.name for source in sources]):
 460            ODHError('All volumes should contain the same fluid.')
 461            return None
 462        if sources[0].fluid.name != fluid.name:
 463            ODHError('New source should have the same fluid as combined.')
 464            return None
 465        volume_NTP = sum([source.N*source.volume for source in sources])
 466        fluid_NTP = ThermState(fluid.name, P=P_NTP, T=T_NTP)
 467        phys_volume = volume_NTP*fluid_NTP.Dmass / fluid.Dmass
 468        return Source(name, fluid, phys_volume, N=N, PFD_isol_valve=PFD_isol_valve)

 588        combinations of fans working. All fans are expected to have the same
 589        volumetric flow.
 590    """
 591
 592    Q_fan: ureg.Quantity
 593    N_fans: int
 594    Test_period: ureg.Quantity
 595    lambda_fan: ureg.Quantity = TABLE_2['Fan']['Failure to run']
 596    const_vent: ureg.Quantity = 0*ureg.ft**3/ureg.min
 597    PFD_ODH: float = PFD_ODH  # Default value for ODH system failure
 598
 599    def fan_flowrates(self):
 600        # TODO add fans with different volumetric rates (see report as well)
 601        fail_rate = self.lambda_fan
 602        fan_flowrates = []
 603        for m in range(self.N_fans+1):
 604            # Probability of exactly m units starting
 605            P_m_fan_work = prob_m_of_n(m, self.N_fans, self.Test_period,
 606                                       fail_rate)
 607            flowrate = self.Q_fan*m
 608            if flowrate == Q_('0 m**3/min'):
 609                flowrate = self.const_vent
 610            fan_flowrates.append((P_m_fan_work, flowrate, m))
 611        return fan_flowrates

 698    def odh(self, sources, power_outage=False):
 699        """Calculate ODH fatality rate for given `Source`s.
 700
 701        For each leak of each source ODH conditions are analyzed and
 702        fatality rates are calculated. The results are collected in
 703        fail_modes list.
 704
 705        Parameters
 706        ----------
 707        sources : List[Source]
 708            Sources affecting the volume.
 709        power_outage : bool
 710            Shows whether there is a power outage is in effect.
 711            Default is no outage.
 712
 713        Returns
 714        -------
 715        None
 716        """
 717        self.fail_modes = []
 718        for source in sources:
 719            for leak in source.leaks:
 720                if not leak._is_const:
 721                    self._fatality_no_power(source, leak, power_outage)
 722                    if not power_outage:
 723                        self._fatality_no_odh(source, leak)
 724                        self._fatality_with_fans(source, leak)
 725                else:
 726                    self._fatality_const_leak(source, leak, power_outage)

 728    def _fatality_no_power(self, source, leak, power_outage):
 729        """Calculate fatality rate in the event of power failure.
 730
 731        Source isolation is the only protection for this case. No ventilation
 732        is available without electric power.
 733
 734        Parameters
 735        ----------
 736        source : Source
 737        leak : Leak
 738            Leak failure rate, volumetric flow rate, event duration, and number
 739            of events.
 740        power_outage : bool
 741            Shows whether there is a power outage is in effect.
 742
 743        Returns
 744        -------
 745        None
 746        """
 747        if power_outage:
 748            PFD_power = 1
 749            tau_event = leak.tau
 750            scenario = 'Planned power outage'
 751        else:
 752            PFD_power = self.power.pfd
 753            tau_event = min(leak.tau, self.power.max_outage)
 754            scenario = 'Power failure'
 755        PFD_isol_valve = source.PFD_isol_valve
 756        P_event = PFD_power * PFD_isol_valve
 757        Q_fan = 0 * ureg.ft**3/ureg.min
 758        self._add_failure_mode(P_event, tau_event, leak.failure_rate, source,
 759                               leak, Q_fan, 0, power_outage, scenario)

 761    def _fatality_no_odh(self, source, leak):
 762        """Calculate fatality rate in the event of ODH system failure.
 763
 764        Power has to be  on for ODH system to fail. Source isolation is
 765        possible for this event. Only ventilation independent from ODH system
 766        is available, e.g., HVAC.
 767
 768        Parameters
 769        ----------
 770        source : Source
 771        leak : Leak
 772            Leak failure rate, volumetric flow rate, event duration, and number
 773            of events.
 774
 775        Returns
 776        -------
 777        None
 778        """
 779        PFD_isol_valve = source.PFD_isol_valve
 780        P_event = (1-self.power.pfd) * self.vent.PFD_ODH * PFD_isol_valve
 781        Q_fan = self.vent.const_vent
 782        tau_event = min(leak.tau, self.vent.Test_period)
 783        scenario = 'ODH system failure'
 784        self._add_failure_mode(P_event, tau_event, leak.failure_rate, source,
 785                               leak, Q_fan, 0, False, scenario)

 787    def _fatality_with_fans(self, source, leak):
 788        """Calculate fatality rate for any number of fans operational.
 789
 790        Power has to be  on, and ODH system operational. Source isolation is
 791        possible for this event. Ventilation independent from ODH system,
 792        e.g., HVAC, is considered for all fans unavailable at the time of
 793        the event.
 794
 795        Parameters
 796        ----------
 797        source : Source
 798        leak : Leak
 799            Leak failure rate, volumetric flow rate, event duration, and number
 800            of events.
 801
 802        Returns
 803        -------
 804        None
 805        """
 806        PFD_isol_valve = source.PFD_isol_valve
 807        # Probability for this group of the events
 808        P_group = (1-self.power.pfd) * (1-self.vent.PFD_ODH) * PFD_isol_valve
 809        tau_event = min(leak.tau, self.vent.Test_period)
 810        for (P_fan, Q_fan, N_fans) in self.vent.fan_flowrates():
 811            P_event = P_group * P_fan  # Probability of the particular event
 812            if N_fans == self.vent.N_fans:
 813                scenario = 'Normal operation'
 814            else:
 815                scenario = 'Fan failure'
 816            self._add_failure_mode(P_event, tau_event,
 817                                   leak.failure_rate, source, leak, Q_fan,
 818                                   N_fans, False, scenario)
 819
 820    def _add_failure_mode(self, P_event, tau_event, failure_rate, source, leak,
 821                          Q_fan, N_fans, power_outage, scenario):
 822        """Add a failure mode.
 823
 824        Parameters
 825        ----------
 826        P_event : float
 827            Probability of a particular state of the system, assuming the leak
 828            has happened. Alternatively, probability of a constant leak or 1.
 829        failure_rate : ureg.Quantity, 1/[time]
 830            Failure rate of the leak. Alternatively, system failure rate for
 831            a constant leak.
 832        tau_event : ureg.Quantity, [time]
 833            Maximum duration of the event.
 834        source : Source
 835            Inert gas source.
 836        leak : Leak
 837            Leak failure rate, volumetric flow rate, event duration, and number
 838            of events.
 839        fluid : ThermState
 840            Fluid conditions at the release location.
 841        Q_fan : ureg.Quantity, [length]^3/[time]
 842            Combined volumetric flow of ODH fans for the event.
 843        N_fans : int
 844            Number of fans operational during the event.
 845        power_outage : bool
 846            Shows whether there is a power outage is in effect.
 847        scenario: str
 848            Name of the response scenario, e.g., power failure, ODH system failure,
 849            normal operation.
 850
 851        Returns
 852        -------
 853        None
 854        """
 855        P_i = failure_rate * P_event
 856        O2_conc = conc_vent(self.volume, leak.q_std, Q_fan, tau_event)
 857        F_i = self._fatality_prob(O2_conc)
 858        phi_i = P_i*F_i
 859        if leak._is_const == True:
 860            N_events = 1
 861        else:
 862            N_events = leak.N_events
 863        self.fail_modes.append(
 864            FailureMode(leak.name, source, leak.fluid, phi_i, O2_conc,
 865                        failure_rate, P_i, F_i, power_outage, leak.q_std,
 866                        tau_event, Q_fan, N_fans, N_events, leak._is_const,
 867                        scenario)
 868        )

 870    def _fatality_const_leak(self, source, leak, power_outage):
 871        """Calculate fatality rates for constant leak case.
 872
 873        Calculate fatality rates for the case of ODH system responding and
 874        fans powered but some of the fans failing on demand.
 875        See wiki for further explanation.
 876        Adds calculation results to the fail_modes list.
 877
 878        Parameters
 879        ----------
 880        source : Source
 881        leak : Leak
 882            Leak failure rate, volumetric flow rate, event duration, and number
 883            of events.
 884        power_outage : bool
 885            Shows whether there is a power outage is in effect.
 886
 887        Returns
 888        -------
 889        None
 890        """
 891
 892        PFD_isol_valve = source.PFD_isol_valve
 893        if power_outage:
 894            tau_outage = leak.tau
 895            Q_fan_outage = 0 * ureg.ft**3/ureg.min  # No ventilation without power
 896            O2_conc_outage = conc_vent(self.volume, leak.q_std, Q_fan_outage, tau_outage)
 897            if O2_conc_outage < 0.195:
 898                raise ConstLeakNoVent(f'Constant leak {leak.name} from '
 899                                    f'{source.name} is not mitigated during '
 900                                    f'power outage in {self.name}.')
 901
 902        # Constant leak and power failure
 903        failure_rate = self.power.lambda_power
 904        # Constant leak is assumed to be hazardous only when
 905        # the isolation valve fails
 906        P_event = PFD_isol_valve * self.power.backup_pfd
 907        Q_fan = 0 * ureg.ft**3/ureg.min  # No ventilation without power
 908        tau_event = min([self.vent.Test_period, self.power.max_outage, leak.tau])
 909        scenario = 'Const leak and power failure'
 910        self._add_failure_mode(P_event, tau_event, failure_rate,
 911                               source, leak, Q_fan, 0, False,
 912                               scenario)
 913
 914        # Constant leak and ODH system failure
 915        failure_rate = lambda_ODH
 916        P_event = 1  # ODH system failure will not trigger isolation valve
 917        # Event can't go undetected for longer than the test period
 918        tau_event = self.vent.Test_period
 919        scenario = 'Const leak and ODH system failure'
 920        # Ventilation is available on ODH system failure
 921        self._add_failure_mode(P_event, tau_event, failure_rate,
 922                               source, leak, self.vent.const_vent, 0, False,
 923                               scenario)
 924
 925        # Fan failure
 926        # If at least 1 fan is down, all fans are considered non-operational
 927        failure_rate = TABLE_2['Fan']['Failure to run']
 928        P_event = PFD_isol_valve
 929        # Event can't go undetected for longer than the test period
 930        tau_event = self.vent.Test_period
 931        scenario = 'Const leak and 1 fan failure'
 932        self._add_failure_mode(P_event, tau_event, failure_rate,
 933                               source, leak, self.vent.const_vent, 0, False,
 934                               scenario)
 935
 936        # Fans operational
 937        # If fatality is possible with just one fan, raise error
 938        Q_1fan = max(self.vent.Q_fan, self.vent.const_vent)
 939        O2_conc_1fan = conc_final(leak.q_std, Q_1fan)
 940        if O2_conc_1fan < 0.195:
 941            raise ConstLeakTooBig('Constant leak reduces O2 concentration to '
 942                                  f'{O2_conc_1fan:.1%} with {Q_1fan:.3g~} '
 943                                  f'vent rate: {leak.name}')

 945    def _fatality_prob(self, O2_conc):
 946        """Calculate fatality probability for given oxygen concentration.
 947
 948        The equation is fitted from the FESHM 4240 plot.
 949
 950        Parameters
 951        ----------
 952        O2_conc : float
 953            Oxygen concentration.
 954
 955        Returns
 956        -------
 957        float
 958            Fatality rate.
 959        """
 960        if O2_conc >= 0.18:  # Lowest oxygen concentration above 18%
 961            Fi = 0
 962        elif O2_conc <= 0.088:  # 8.8% of oxygen is assumed to be 100% fatal
 963            Fi = 1
 964        else:
 965            # Fi formula, reverse engineered using 8.8% and 18% thresholds
 966            Fi = 10**(6.5-76*O2_conc)
 967        return Fi

1125def prob_m_of_n(m, n, T, l):
1126    """Calculate the probability of m out of n units working.
1127
1128    Calculation is done using binomial distribution.
1129
1130    Parameters
1131    ----------
1132    m : int
1133        Number of units working.
1134    n : int
1135        Total number of units.
1136    T : ureg.Quantity, [time]
1137        Test period
1138    l : ureg.Quantity, 1/[time]
1139        Failure rate (\\lambda) of a fan
1140
1141    Returns
1142    -------
1143    float
1144        Probability of m out of n units working.

1151def conc_vent(V, R, Q, t):
1152    """Calculate the oxygen concentration at the end of the event.
1153
1154    As defined by FESHM 4240 6.1.A, Cases A, B, and C.
1155
1156    Parameters
1157    ----------
1158    V : ureg.Quantity, [length]^3
1159        Volume of the confined space.
1160    R : ureg.Quantity, [length]^3/[time]
1161        Volumetric spill rate into confined space.
1162    Q : ureg.Quantity, [length]^3/[time]
1163        Volumetric ventilation rate of fan(s); positive value corresponds
1164        to blowing air into the confined space, negative - drawing contaminated
1165        air outside.
1166    t : ureg.Quantity, [time]
1167        time, beginning of release is at `t` = 0.
1168
1169    Returns
1170    -------
1171    float
1172        Oxygen concentration.
1173    """
1174    if Q > 0:
1175        C = 0.21/(Q+R) * (Q+R*math.e**-((Q+R)/V*t))
1176    elif abs(Q) <= R:
1177        C = 0.21*math.e**-(R/V*t)
1178    elif abs(Q) > R:
1179        C = 0.21*(1-R/abs(Q)*(1-math.e**-(abs(Q)*t/V)))
1180    return float(C)

 947        """Calculate fatality probability for given oxygen concentration.

 955
 956        Returns
 957        -------
 958        float
 959            Fatality rate.
 960        """
 961        if O2_conc >= 0.18:  # Lowest oxygen concentration above 18%
 962            Fi = 0
 963        elif O2_conc <= 0.088:  # 8.8% of oxygen is assumed to be 100% fatal
 964            Fi = 1
 965        else:
